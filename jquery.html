Trucos Jquery necesarios

Cargar página web externa (AJAX)

traer página web externa (se debe estar en un servidor local o web)


	// traer web externa
  	$(document).ready(function(){
	    $("#nombre-de-div").load("nombre-pagina.html");
	});

	<!-- en este div se cargara la pagina -->
	<div id="nombre-de-div"></div>

Cargar contenido externo con un (button)


	<script>
	// cargar contenido externo con un (button)
	$(document).ready(function(){
	  $("button").click(function(){
	    $("#div1").load("demo_test.txt");
	  });
	});
	</script>
	<body>
		<button>boton</button>
		<!-- en este div se cargara la pagina -->
		<div id="div1"></div>
	</body>
