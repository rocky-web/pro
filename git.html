Trucos GIT necesarios

Algunos Comandos GIT LOG 

	// Listar todos los commit realizados
 	git log


	// Listar todos los commit realizados en 1 sola linea
	git log --oneline


	// Listar todos los commit realizados en 1 sola linea y ademas mostrar los nombres de archivos 
	// modificados en el commit con --name-only. Con -5 decimos que muestre solo las 5 ultimas lineas
	git log --oneline --name-only -5 


	// Saber que nombres de archivos se editaron en commit especifico
	git log --name-only c4a087a
Algunos Comandos GIT DIFF 

	// ver los cambios del codigo del ultimo commit
	git diff


  	// ver los cambios del codigo de un commit especificado hasta el ultimo commit guardado
  	git diff mi-commit


	// ver los cambios del codigo de un archivo desde el commit seleccionado hasta el ultimo commit
	git diff c4a087a 'ruta git.txt'


	// Listar los nombres de archivos que se modificaron desde el commit especificado hasta el ultimo commit
	git diff --name-only c939d35



	// Forzar cambios en git local a git remoto (Github). En caso de que hemos guardado un  cambio en github y luego lo quiero deshacer. 
	// -f es igual a --force
	git push origin -f
